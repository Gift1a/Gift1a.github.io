import base64


def encode(message):
    s = ''
    for i in message:
        x = i ^ 32
        x += 16
        s += chr(x)
        return (base64.b64encode(s.encode('utf-8')))


def enCodeAgain(string, space):
    s = ''
    string = str(string, 'utf-8')
    for i in range(0, space):
        for j in range(i, len(string), space):
            if j < len(string):
                s += string[j]
    return s
